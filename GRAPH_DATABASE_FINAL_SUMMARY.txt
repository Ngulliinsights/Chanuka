import {
  initializeNeo4jDriver,
  syncPersonToGraph,
  createSponsorshipRelationship,
} from '@shared/database/graph';

// Initialize
initializeNeo4jDriver({

  username: process.env.NEO4J_USERNAME,

});

await syncPersonToGraph(person);
